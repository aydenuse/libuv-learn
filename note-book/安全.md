      const self = this;
      const args = Array.prototype.slice.call(arguments, 1);
      const fn = function () {
          const thisArgs = Array.prototype.slice.call(arguments);
          context = this instanceof fn ? this : context;

      }
      fn.prototype = this.prototype;
      return fn;
  }
